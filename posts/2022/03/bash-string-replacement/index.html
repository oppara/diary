目次

はじめに#

よく使う変数内文字列置換を忘れるのでのメモ。

例#

~/tmp/hoge.sh

#!/bin/bash

echo "$0"

# basename
echo "${0##*/}"

# 拡張子取得
echo "${0##*\.}"

# dirname
echo "${0%/*}"

# 拡張子以外
echo "${0%.*}"

output

/Users/oppara/tmp/hoge.sh
hoge.sh
sh
/Users/oppara/tmp
/Users/oppara/tmp/hoge
